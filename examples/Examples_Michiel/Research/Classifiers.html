Classifying good from bad spectra¶

In this notebook we will try to separate 'good' from 'bad' spectra using some off-the-shelf classifiers.

Loading and preparing the data¶

In [2]:
dataset = loadmat('dataClassification.mat')
In [3]:
positive_spectra = dataset['dataSelectedTotal']
negative_spectra = dataset['dataBadTotal']

spectra_raw = np.vstack([positive_spectra, negative_spectra])
spectra_sav = savgol_filter(spectra_raw, window_length=5, polyorder=3, deriv=1)
labels = np.array([1]*positive_spectra.shape[0] + [0]*negative_spectra.shape[0])

spectra_raw = spectra_raw[:,10:-10]
spectra_sav = spectra_sav[:,10:-10]

freqs = np.arange(spectra_raw.shape[1])

fig, (ax0, ax1) = plt.subplots(nrows=2)

    if labels[i] == 1:

    ax0.plot(freqs, spectra_raw[i,:], c=col, alpha=0.5)

random_forest = GridSearchCV(RandomForestClassifier(n_jobs=-1), param_grid = {'n_estimators':[10, 50, 100]})
adaboost = GridSearchCV(AdaBoostClassifier(), param_grid = {'n_estimators':[10, 25, 50], 'learning_rate':[0.5, 0.75, 1]})
svc = GridSearchCV(SVC(), param_grid={'C' : [0.1, 1, 10]})

algorithms = {'RF':random_forest, 'adaboost':adaboost, 'SVC':svc}
#algorithms = {'SVC':svc}

performance = {k:[] for k in algorithms.keys()}

kfold_generator = KFold(len(labels), n_folds=4, shuffle=True)
In [18]:
for train, test in kfold_generator:
    for name, alg in algorithms.iteritems():
        print name
        alg.fit(spectra_sav[train], labels[train])
        performance[name].append(np.mean(labels[test] == alg.predict(spectra_sav[test])))
